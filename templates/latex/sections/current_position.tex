~< extends "section.tex" >~

~< block body >~
~< for item in items >~
    ~< if item.pdf_logo ->~
    \raisebox{-0.2\height}{\includegraphics[height=.9em]{<< item.pdf_logo>>}}
    ~< endif >~
    ~<- if item.title ->~
    \textbf{<< item.title >>}, \emph{<< item.place >>}
    ~<- else ->~
    \textbf{<< item.place>>}
    ~<- endif ->~
    ~<- if item.location ->~
    , << item.location >>
    ~< endif >~
    ~< if item.inline_detail >~
    {\footnotesize \color{gray} (<< item.inline_detail >>)}
    ~< endif >~
    \hfill << item.dates >> \\
    ~< if item.details >~
    \begin{itemize}[leftmargin=1.5em, nosep]
    ~< for detail in item.details ->~
        \item <<detail>>
    ~< endfor >~
    \end{itemize}
    ~< endif >~
~< endfor >~
\vspace{-1mm}
~< endblock body >~
