~< extends "section.tex" >~

~< block body >~
~< for item in items >~
    ~< if item.pdf_logo ->~
    \raisebox{-0.2\height}{\includegraphics[height=.9em]{<< item.pdf_logo>>}}
    ~< endif >~
    ~<- if item.title ->~
    \textbf{<< item.title >>}, \emph{<< item.place >>}
    ~<- else ->~
    \textbf{<< item.place>>}
    ~<- endif ->~
    ~<- if item.location ->~
    , << item.location >>
    ~< endif >~
    ~< if item.inline_detail >~
    {\footnotesize \color{gray} (<< item.inline_detail >>)}
    ~< endif >~
    \hfill << item.dates >> \\
    ~< if item.details >~
    {\small <<item.details>>} \\
    ~< endif >~
~< endfor >~
\vspace{-8mm}
~< endblock body >~
